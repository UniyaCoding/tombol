Lua Script - Carousel Toggle


local ProximityPrompt = script.Parent.ProximityPrompt
local HorsesModel = workspace.Carousel.Horses
local RotatingScript = HorsesModel.Script
local toggle = false
ProximityPrompt.ActionText = "Activate"

ProximityPrompt.Triggered:Connect(function()
  if not toggle then
    toggle = true
    ProximityPrompt.ActionText = "Deactivate"
    RotatingScript.Enabled = true
  else
    toggle = false
    ProximityPrompt.ActionText = "Activate"
    RotatingScript.Enabled = false
  end
end)
